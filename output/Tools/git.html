初始化

git init

添加文件

git add fileName
git add .

git commit -m message

# 文件删除
git rm

文件上传、下载

git pull

查看状态

# 仓库当前的状态
git status

# 仓库当前log
git log

# reset之后，找不到新版本的 commit id时，可以使用找回log
git reflog

# 查看工作区和版本库里面最新版本的区别
git diff HEAD -- readme.txt

版本回退

# 撤销修改
git checkout -- readme.txt

# 撤回指定版本
git reset --hard HEAD

添加远程仓库

# 克隆仓库 
git clone git@github.com:michaelliao/gitskills.git

# 绑定远程仓库
git remote add origin git@github.com:michaelliao/learngit.git

#  推送到master分支
 git push -u origin master

分支操作

# 查看当前分支
git branch

# 创建 dev 分支
git branch dev

# 切换到 dev 分支
git checkout dev

# 创建 dev 分支，然后切换到 dev 分支
git checkout -b dev

# 合并dev分支到当前分支
git merge dev

# 删除 dev 分支
git branch -d dev

# 查看分支合并图
git log --graph

其他操作

# Git 显示颜色
git config --global color.ui true
